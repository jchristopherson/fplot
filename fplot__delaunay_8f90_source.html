6  module subroutine d2d_init(this, x, y, err)
7  ! Arguments
8  class(delaunay_tri_2d), intent(inout) :: this
9  real(real64), intent(in), dimension(:) :: x, y
10  class(errors), intent(inout), target, optional :: err
11 
12  ! Local Variables
13  class(errors), pointer :: errmgr
14  type(errors), target :: deferr
15  character(len = 256) :: errmsg
16  integer(int32) :: i, npts, ntri, flag
17  real(real64), allocatable, dimension(:,:) :: nodexy
18  integer(int32), allocatable, dimension(:,:) :: trinode, trinbr
19 
20  ! Initialization
21  if (present(err)) then
22  errmgr => err
23  else
24  errmgr => deferr
25  end if
26  npts = size(x)
27 
28  ! Input Check
29  if (size(y) /= npts) then
30  write(errmsg, '(AI0AI0A)') &
31  "Expected the y-coordinate array to have ", npts, &
32  " elements, but found ", size(y), " instead."
33  call errmgr%report_error("d2d_init", trim(errmsg), &
34  plot_array_size_mismatch_error)
35  return
36  end if
37 
38  ! Clean up incase of an existing triangulation
39  if (allocated(this%m_x)) deallocate(this%m_x)
40  if (allocated(this%m_y)) deallocate(this%m_y)
41  if (allocated(this%m_indices)) deallocate(this%m_indices)
42 
43  ! Allocate workspace arrays for the triangulation
44  allocate(nodexy(2, npts), stat = flag)
45  if (flag == 0) allocate(trinode(3, 2 * npts), stat = flag)
46  if (flag == 0) allocate(trinbr(3, 2 * npts), stat = flag)
47  if (flag /= 0) go to 100
48 
49  ! Generate the points list
50  do i = 1, npts
51  nodexy(1,i) = x(i)
52  nodexy(2,i) = y(i)
53  end do
54 
55  ! Compute the triangulation
56  call r8tris2(npts, nodexy, ntri, trinode, trinbr)
57 
58  ! Populate the remainder of the object
59  allocate(this%m_x(npts), stat = flag)
60  if (flag == 0) allocate(this%m_y(npts), stat = flag)
61  if (flag == 0) allocate(this%m_indices(ntri, 3), stat = flag)
62 
63  do i = 1, npts
64  this%m_x(i) = nodexy(1,i)
65  this%m_y(i) = nodexy(2, i)
66  end do
67 
68  do i = 1, ntri
69  this%m_indices(i,:) = trinode(:,i)
70  end do
71 
72  ! End
73  return
74 
75  ! Memory Error Handler
76  100 continue
77  call errmgr%report_error("d2d_init", "Insufficient memory available.", &
78  plot_out_of_memory_error)
79  return
80  end subroutine

139  ! Arguments

173 ! ------------------------------------------------------------------------------
174  ! Determine if a point lies within a triangle.
175  ! https://stackoverflow.com/questions/2049582/how-to-determine-if-a-point-is-in-a-2d-triangle
176  ! https://en.wikipedia.org/wiki/Barycentric_coordinate_system
177  pure elemental function point_inside_triangle(x1, y1, x2, y2, x3, y3, &
178  x, y) result(rst)
179  ! Arguments
180  real(real64), intent(in) :: x1, y1, x2, y2, x3, y3, x, y
181  logical :: rst
182 
183  ! Local Variables
184  real(real64) :: lambda1, lambda2, dT
185 
186  ! Initialization
187  dt = (y2 - y3) * (x1 - x3) + (x3 - x2) * (y1 - y3)
188  lambda1 = ((y2 - y3) * (x - x3) + (x3 - x2) * (y - y3)) / dt
189  lambda2 = ((y3 - y1) * (x - x3) + (x1 - x3) * (y - y3)) / dt
190 
191  ! The point is within the triangle if:
192  ! 0 <= lambda1 <= 1
193  ! 0 <= lambda2 <= 1
194  ! 0 <= lambda1 + lambda2 <= 1
195  rst = (lambda1 <= 1.0d0 .and. lambda1 >= 0.0d0) .and. &
196  (lambda2 <= 1.0d0 .and. lambda2 >= 0.0d0) .and. &
197  (lambda1 + lambda2 >= 0.0d0 .and. lambda1 + lambda2 <= 1.0d0)
198  end function
